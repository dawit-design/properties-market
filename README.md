|__ assets/

      |__ mockups

      |__ Header.jsx
      |__ Modal.jsx

      |__ PropertyContainer/
        |__ PropertyContainer.jsx
|__ layouts/
        |__ Layout.jsx
|__ screens/
      |__ Login/
        |__ Login.jsx
      |__ Register/

      |__ PropertyCreate/

      |__ PropertyDetail/
        |__ PropertyDetail.jsx
